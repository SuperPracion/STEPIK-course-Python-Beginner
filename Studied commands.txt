print() - вывод данных
input() - ввод данных
print(sep=*) - разделение всех выводимых данных
print(end=*) - изменение окончания строки
+ - сложение
- - вычитание
/ - деление
* - умножение 
= - приравнивание
** - возведение в степень
// - деление без остатка с округлением в меньшую сторону
% - деление с остатком
if - else - условный оператор
== - логическое сравнение
!= - логическое сравнение с не
and - логическое И
or - логическое или 
not - логическое не
int() - числовой тип данных
float() - числами с плавающей точкой
max(), min(), abs() - функции: максимума(диапазон), минимум(диапазон), модуль(число)
str() - текстовый тип данных
len() - функция возвращающая длину
in - оператор вложенности/членства
''' ''' - Многострочный текст, используя внутри "." - переносим строку.
ceil() - округление числа вверх
floor() - округление числа вниз
pow(x, n) - возведение числа x в степень n
+= - x = x + n (упрощенный варианта само суммы)
-= - x = x - n (упрощённый вариант само разницы)
*= - x = x * n (упрощённый вариант само произведения)
/= - x = x / n (упрощённый вариант само деления)
//= - x = x // n (упрощенный вариант само деления без остатка)
%= - x = x % n (упрощенный вариант само деления с остатком)
break - оператор прерывания цикла
continue - оператор возврата в тело цикла
for/while-else - необязательный блок выполняющийся только после штатного завершения цикла
capitalize() - метод приводящей строку к текстовому формату. 'Тут был я'
swapcase() - метод реверсирующий регистр у всех символов в строке
title() - метод приводящий строку к формату 'Новое Слово Большая Буква'
lower() - метод приводящий все символы в нижний регистр
upper() - метод приводящий все символы в строке к верхнему регистру
count(<sub>, <start>, <end>) - метод считает количество повторений в строке аргумента <sub> (вернёт число)
startswith(<suffix>, <start>, <end>) - метод проверяет начинается ли строка со строки <suffix> (вернёт лог. значение)
endswith(<suffix>, <start>, <end>) - метод проверяет оканчивается ли строка строкой <suffix> (вернёт лог. значение)
find(<sub>), rfind(<sub>) - метод производит поиск <sub> до первого совпадения (вернёт индекс или -1)
index(<sub>), rindex(<sub>) -метод аналогичен fined, но выдаёт ошибку если нет совпадений
strip() - метод удаляет пробелы в начале и в конце строки
lstrip() - метод удалит только пробелы в начале строки
rstrip() - метод удалит только пробелы в конце строки
replace(<old>, <new>, <count>) - метод заменит все старые <old> значения на новые <new> сколько раз <count>
isalnum() - метод определяет состоит ли строка только из буквенно-цифирных символов, пустая или есть посторонние символы - False
isalpha() - метод определяет состоит ли строка только из буквенных символов, иначе - False
isdigit() - метод определяет состоит ли строка только из цифровых символов, иначе - False
islower() - метод определяет являются ли все символы строки нижнего регистра + цифры и символы, иначе - False
isupper() - метод определяет являются ли все символы строки верхнего регистра + цифры и символы, иначе - False
isspace() - метод определяет является ли строка только пробелами, иначе - False
ord() - функция определяет кодировку символа
chr() - функция переводит код в символ
list() - не массив, а список. Тип данных хранящий в себе различные последовательности из элементов
