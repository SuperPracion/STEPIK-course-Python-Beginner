print() - вывод данных
input() - ввод данных
print(sep=*) - разделение всех выводимых данных
print(end=*) - изменение окончания строки
+ - сложение
- - вычитание
/ - деление
* - умножение 
= - приравнивание
** - возведение в степень
// - деление без остатка с округлением в меньшую сторону
% - деление с остатком
if - else - условный оператор
== - логическое сравнение
!= - логическое сравнение с не
and - логическое И
or - логическое или 
not - логическое не
int() - числовой тип данных
float() - числами с плавающей точкой
max(), min(), abs() - функции: максимума(диапазон), минимум(диапазон), модуль(число)
str() - текстовый тип данных
len() - функция возвращающая длину
in - оператор вложенности/членства
''' ''' - Многострочный текст, используя внутри "." - переносим строку.
ceil() - округление числа вверх
floor() - округление числа вниз
pow(x, n) - возведение числа x в степень n
+= - x = x + n (упрощенный варианта самосуммы)
-= - x = x - n (упрощённый вариант саморазницы)
*= - x = x * n (упрощённый вариант самопроизведения)
/= - x = x / n (упрощённый вариант самоделения)
//= - x = x // n (упрощенный вариант самоделения без остатка)
%= - x = x % n (упрощенный вариант самоделения с остатком)
